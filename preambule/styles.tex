\makeatletter
\newcommand*{\epigramme}[0]{
	\begin{list}{}{
		\leftmargin 0.25\linewidth
		\listparindent \z@
		\listparindent \z@
                     \itemindent    \listparindent
                     \rightmargin   0.05\linewidth
                     \parsep        \z@ \@plus\p@
		}
		\item\ifthenelse{\equal{\epi}{}}{}{
		\enquote{\epi}}
	
	\begin{flushright}{\small\episource}\end{flushright}
	\end{list}

	\def\episource{}
	\def\epi{}
}
